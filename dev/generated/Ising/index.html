using CondensedMatterSOS
@spin σ[1:2]
function hamiltonian(σ)
    σx, σy, σz = σ
    N = length(σx)
    return -sum(σx[n] * σx[n+1] for n in 1:(N-1)) - σx[N] * σx[1] - sum(σz)
end
hamiltonian(σ)

function hamiltonian_energy(N, maxdegree, solver; kws...)
    @spin σ[1:N]
    H = 1.0 * hamiltonian(σ)
    energy(H, maxdegree, solver; kws...)
end
